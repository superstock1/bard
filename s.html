import requests from bs4 import BeautifulSoup # 네이버 API 키를 입력합니다. naver_api_key = "1xHCVe_XENpNPiLZmqEr" # 검색할 키워드를 입력합니다. keyword = "현대차" # 네이버 뉴스 API를 호출하여 뉴스 기사 목록을 가져옵니다. response = requests.get("https://openapi.naver.com/v1/search/news.json?query=" + keyword + "&start=1&display=10&sort=date", headers={"Authorization": "Bearer " + naver_api_key}) # 뉴스 기사 목록을 파싱합니다. soup = BeautifulSoup(response.content, "html.parser") # 뉴스 기사 목록을 출력합니다. for news in soup.find_all("item"): print(news.find("title").text) print(news.find("link").text) print(news.find("description").text)
